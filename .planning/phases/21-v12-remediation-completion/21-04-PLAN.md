```rust
/// ReadableStream constructor
fn readable_stream_constructor(
    scope: &mut v8::HandleScope,
    args: v8::FunctionCallbackArguments,
    mut retval: v8::ReturnValue,
) {
    let this = args.this();
    let instance = if let Some(obj) = this.to_object(scope) {
        obj
    } else {
        v8::Object::new(scope)
    };
    
    // Store underlying source if provided
    if args.length() > 0 {
        let source = args.get(0);
        let source_key = v8::String::new(scope, "__source").unwrap();
        instance.set(scope, source_key.into(), source.into());
    }
    
    // Initialize state
    let state_key = v8::String::new(scope, "__state").unwrap();
    let state_val = v8::String::new(scope, "readable").unwrap();
    instance.set(scope, state_key.into(), state_val.into());
    
    retval.set(instance.into());
}

/// getReader() method
fn readable_stream_get_reader(
    scope: &mut v8::HandleScope,
    args: v8::FunctionCallbackArguments,
    mut retval: v8::ReturnValue,
) {
    let this = args.this();
    
    // Create reader instance
    let global = scope.get_current_context().global(scope);
    let reader_key = v8::String::new(scope, "ReadableStreamDefaultReader").unwrap();
    
    if let Some(reader_ctor) = global.get(scope, reader_key.into()) {
        if let Some(reader_fn) = reader_ctor.to_object(scope) {
            if let Some(reader) = reader_fn.cast::<v8::Function>().new_instance(scope, &[this.into()]) {
                retval.set(reader.into());
                return;
            }
        }
    }
    
    retval.set(v8::null(scope).into());
}
```

```rust
        if let Some(stream_obj) = stream.to_object(scope) {
            if let Some(source) = stream_obj.get(scope, source_key.into()) {
                if source.is_object() {
                    // Call start() if not already called

                    // This is simplified - full implementation needs queue management
                    
                    // Return { value, done } object
                    let result = v8::Object::new(scope);
```

### Step 5: Bind All Stream APIs

**File:** `src/runtime/streams.rs`

```rust
pub fn bind_streams_api(
    scope: &mut v8::ContextScope<v8::HandleScope>,
    context: v8::Local<v8::Context>,
) {
    let global = context.global(scope);
    
    // ReadableStream
    let rs_key = v8::String::new(scope, "ReadableStream").unwrap();
    let rs_ctor = v8::Function::new(scope, readable_stream_constructor).unwrap();
    if let Some(rs_obj) = rs_ctor.to_object(scope) {
        let proto_key = v8::String::new(scope, "prototype").unwrap();
        let proto = v8::Object::new(scope);
        bind_stream_method(scope, proto, "getReader", readable_stream_get_reader);
        rs_obj.set(scope, proto_key.into(), proto.into());
    }
    global.set(scope, rs_key.into(), rs_ctor.into());
    
    // ReadableStreamDefaultReader
    let reader_key = v8::String::new(scope, "ReadableStreamDefaultReader").unwrap();
    let reader_ctor = v8::Function::new(scope, readable_stream_default_reader_constructor).unwrap();
    if let Some(reader_obj) = reader_ctor.to_object(scope) {
        let proto_key = v8::String::new(scope, "prototype").unwrap();
        let proto = v8::Object::new(scope);
        bind_stream_method(scope, proto, "read", reader_read_callback);
        bind_stream_method(scope, proto, "releaseLock", reader_release_lock_callback);
        reader_obj.set(scope, proto_key.into(), proto.into());
    }
    global.set(scope, reader_key.into(), reader_ctor.into());
    
    // WritableStream
    let ws_key = v8::String::new(scope, "WritableStream").unwrap();
    let ws_ctor = v8::Function::new(scope, writable_stream_constructor).unwrap();
    if let Some(ws_obj) = ws_ctor.to_object(scope) {
        let proto_key = v8::String::new(scope, "prototype").unwrap();
        let proto = v8::Object::new(scope);
        bind_stream_method(scope, proto, "getWriter", writable_stream_get_writer);
        ws_obj.set(scope, proto_key.into(), proto.into());
    }
    global.set(scope, ws_key.into(), ws_ctor.into());
    
    // WritableStreamDefaultWriter
    let writer_key = v8::String::new(scope, "WritableStreamDefaultWriter").unwrap();
    let writer_ctor = v8::Function::new(scope, writable_stream_default_writer_constructor).unwrap();
    if let Some(writer_obj) = writer_ctor.to_object(scope) {
        let proto_key = v8::String::new(scope, "prototype").unwrap();
        let proto = v8::Object::new(scope);
        bind_stream_method(scope, proto, "write", writer_write_callback);
        bind_stream_method(scope, proto, "close", writer_close_callback);
        bind_stream_method(scope, proto, "releaseLock", writer_release_lock_callback);
        writer_obj.set(scope, proto_key.into(), proto.into());
    }
    global.set(scope, writer_key.into(), writer_ctor.into());
}
```
