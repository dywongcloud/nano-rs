```rust
//! Timer functions for JavaScript (setTimeout, setInterval, clearTimeout, clearInterval)

use std::collections::HashMap;
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Mutex;

/// Global timer ID counter
static TIMER_ID_COUNTER: AtomicU64 = AtomicU64::new(1);

/// Timer storage - maps timer ID to abort handle
lazy_static::lazy_static! {
    static ref TIMERS: Mutex<HashMap<u64, tokio::task::AbortHandle>> = Mutex::new(HashMap::new());
}

/// Generate unique timer ID
fn next_timer_id() -> u64 {
    TIMER_ID_COUNTER.fetch_add(1, Ordering::SeqCst)
}

/// Store timer handle
fn store_timer(id: u64, handle: tokio::task::AbortHandle) {
    if let Ok(mut timers) = TIMERS.lock() {
        timers.insert(id, handle);
    }
}

/// Remove and abort timer
fn remove_timer(id: u64) -> bool {
    if let Ok(mut timers) = TIMERS.lock() {
        if let Some(handle) = timers.remove(&id) {
            handle.abort();
            return true;
        }
    }
    false
}
```

/// setTimeout(callback, delay_ms, ...args)
fn set_timeout_callback(
    scope: &mut v8::HandleScope,
    args: v8::FunctionCallbackArguments,
    mut retval: v8::ReturnValue,
) {
    // Get callback function
    let callback = if args.length() > 0 {
        let cb = args.get(0);
        if cb.is_function() {
            v8::Global::new(scope, cb.cast::<v8::Function>())
        } else {
            retval.set(v8::Number::new(scope, 0.0).into());
            return;
        }
    } else {
        retval.set(v8::Number::new(scope, 0.0).into());
        return;
    };
    
    // Get delay (default 0)
    let delay_ms = if args.length() > 1 {
        let delay = args.get(1);
        delay.to_int32(scope).map(|i| i.value() as u64).unwrap_or(0)
    } else {
        0
    };
    
    // Collect additional arguments
    let additional_args: Vec<v8::Global<v8::Value>> = (2..args.length())
        .map(|i| v8::Global::new(scope, args.get(i)))
        .collect();
    
    // Generate timer ID
    let timer_id = next_timer_id();
    
    // Create async task for the timer
    let isolate_ptr = scope as *mut v8::HandleScope as *mut std::ffi::c_void;
    
    let handle = tokio::spawn(async move {
        // Wait for delay
        tokio::time::sleep(tokio::time::Duration::from_millis(delay_ms)).await;
        
        // Execute callback - this needs isolate access
        // For now, just log that timer fired
        tracing::debug!("Timer {} fired after {}ms", timer_id, delay_ms);
        
        // Note: Actually calling the JS callback requires more complex
        // isolate management. For v1.2.0, we may just track timer state.
    });
    
    // Store abort handle
    store_timer(timer_id, handle.abort_handle());
    
    // Return timer ID
    retval.set(v8::Number::new(scope, timer_id as f64).into());
}

/// setInterval(callback, interval_ms, ...args)
fn set_interval_callback(
    scope: &mut v8::HandleScope,
    args: v8::FunctionCallbackArguments,
    mut retval: v8::ReturnValue,
) {
    // Get callback function
    let callback = if args.length() > 0 {
        let cb = args.get(0);
        if cb.is_function() {
            v8::Global::new(scope, cb.cast::<v8::Function>())
        } else {
            retval.set(v8::Number::new(scope, 0.0).into());
            return;
        }
    } else {
        retval.set(v8::Number::new(scope, 0.0).into());
        return;
    };
    
    // Get interval (default 0)
    let interval_ms = if args.length() > 1 {
        let interval = args.get(1);
        interval.to_int32(scope).map(|i| i.value() as u64).unwrap_or(0)
    } else {
        0
    };
    
    // Generate timer ID
    let timer_id = next_timer_id();
    
    // Create repeating task
    let handle = tokio::spawn(async move {
        let mut interval = tokio::time::interval(tokio::time::Duration::from_millis(interval_ms));
        loop {
            interval.tick().await;
            tracing::debug!("Interval {} fired", timer_id);
            // Note: Actually calling JS callback requires isolate access
        }
    });
    
    // Store abort handle
    store_timer(timer_id, handle.abort_handle());
    
    // Return interval ID
    retval.set(v8::Number::new(scope, timer_id as f64).into());
}

### Step 4: Implement clearTimeout and clearInterval

/// clearTimeout(timerId)
fn clear_timeout_callback(
    scope: &mut v8::HandleScope,
    args: v8::FunctionCallbackArguments,
    mut retval: v8::ReturnValue,
) {
    let timer_id = if args.length() > 0 {
        args.get(0).to_int32(scope).map(|i| i.value() as u64).unwrap_or(0)
    } else {
        0
    };
    
    remove_timer(timer_id);
    retval.set(v8::undefined(scope).into());
}

/// clearInterval(intervalId) - same as clearTimeout
fn clear_interval_callback(
    scope: &mut v8::HandleScope,

    // Same implementation as clearTimeout

    } else {
        0
    };

### Step 5: Bind Timer Functions

```rust
/// Bind all timer functions to global scope
pub fn bind_timers_api(
    scope: &mut v8::ContextScope<v8::HandleScope>,
    context: v8::Local<v8::Context>,
) {
    let global = context.global(scope);
    
    // setTimeout
    let set_timeout_key = v8::String::new(scope, "setTimeout").unwrap();
    let set_timeout = v8::Function::new(scope, set_timeout_callback).unwrap();
    global.set(scope, set_timeout_key.into(), set_timeout.into());
    
    // setInterval
    let set_interval_key = v8::String::new(scope, "setInterval").unwrap();
    let set_interval = v8::Function::new(scope, set_interval_callback).unwrap();
    global.set(scope, set_interval_key.into(), set_interval.into());
    
    // clearTimeout
    let clear_timeout_key = v8::String::new(scope, "clearTimeout").unwrap();
    let clear_timeout = v8::Function::new(scope, clear_timeout_callback).unwrap();
    global.set(scope, clear_timeout_key.into(), clear_timeout.into());
    
    // clearInterval
    let clear_interval_key = v8::String::new(scope, "clearInterval").unwrap();
    let clear_interval = v8::Function::new(scope, clear_interval_callback).unwrap();
    global.set(scope, clear_interval_key.into(), clear_interval.into());
}
```
