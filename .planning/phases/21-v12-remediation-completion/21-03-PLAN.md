/// Bind URLSearchParams constructor and methods
pub fn bind_url_search_params_api(
    scope: &mut v8::ContextScope<v8::HandleScope>,
    context: v8::Local<v8::Context>,
) {
    let global = context.global(scope);
    
    // Create URLSearchParams constructor
    let usp_key = v8::String::new(scope, "URLSearchParams").unwrap();
    let usp_ctor = v8::Function::new(scope, url_search_params_constructor).unwrap();
    
    // Add prototype methods
    if let Some(usp_obj) = usp_ctor.to_object(scope) {
        let prototype_key = v8::String::new(scope, "prototype").unwrap();
        let prototype = v8::Object::new(scope);
        
        bind_url_method(scope, prototype, "get", usp_get_callback);
        bind_url_method(scope, prototype, "set", usp_set_callback);
        bind_url_method(scope, prototype, "has", usp_has_callback);
        bind_url_method(scope, prototype, "delete", usp_delete_callback);
        bind_url_method(scope, prototype, "append", usp_append_callback);
        bind_url_method(scope, prototype, "toString", usp_tostring_callback);
        bind_url_method(scope, prototype, "entries", usp_entries_callback);
        
        usp_obj.set(scope, prototype_key.into(), prototype.into());
    }
    
    global.set(scope, usp_key.into(), usp_ctor.into());
}

### Step 2: Update URL Constructor

**File:** `src/runtime/url.rs`

**Modify URL constructor to create searchParams:**

```rust
fn url_constructor_callback(
    scope: &mut v8::HandleScope,
    args: v8::FunctionCallbackArguments,
    mut retval: v8::ReturnValue,
) {
    let this = args.this();
    let instance = if let Some(obj) = this.to_object(scope) {
        obj
    } else {
        v8::Object::new(scope)
    };
    
    // Parse URL from arguments
    let url_string = if args.length() > 0 {
        let arg = args.get(0);
        arg.to_string(scope)
            .map(|s| s.to_rust_string_lossy(scope))
            .unwrap_or_default()
    } else {
        String::new()
    };
    
    // Parse URL
    if let Ok(url) = url::Url::parse(&url_string) {
        // Set URL components
        let href_key = v8::String::new(scope, "href").unwrap();
        let href_val = v8::String::new(scope, url.as_str()).unwrap();
        instance.set(scope, href_key.into(), href_val.into());
        
        let protocol_key = v8::String::new(scope, "protocol").unwrap();
        let protocol_val = v8::String::new(scope, &format!("{}:", url.scheme())).unwrap();
        instance.set(scope, protocol_key.into(), protocol_val.into());
        
        let host_key = v8::String::new(scope, "host").unwrap();
        let host_val = v8::String::new(scope, url.host_str().unwrap_or("")).unwrap();
        instance.set(scope, host_key.into(), host_val.into());
        
        let pathname_key = v8::String::new(scope, "pathname").unwrap();
        let pathname_val = v8::String::new(scope, url.path()).unwrap();
        instance.set(scope, pathname_key.into(), pathname_val.into());
        
        // Create URLSearchParams for searchParams property
        let search_key = v8::String::new(scope, "search").unwrap();
        let search_val = v8::String::new(scope, url.query().map(|q| format!("?{}", q)).unwrap_or_default().as_str()).unwrap();
        instance.set(scope, search_key.into(), search_val.into());
        
        // Create URLSearchParams instance
        let usp_key = v8::String::new(scope, "URLSearchParams").unwrap();
        if let Some(usp_ctor) = scope.get_current_context().global(scope).get(scope, usp_key.into()) {
            if let Some(usp_fn) = usp_ctor.to_object(scope) {
                if let Some(usp_func) = usp_fn.cast::<v8::Function>().new_instance(scope, &[search_val.into()]) {
                    let search_params_key = v8::String::new(scope, "searchParams").unwrap();
                    instance.set(scope, search_params_key.into(), usp_func.into());
                }
            }
        }
    }
    
    retval.set(instance.into());
}
```
