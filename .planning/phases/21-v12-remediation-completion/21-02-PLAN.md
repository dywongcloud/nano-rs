**File:** `src/runtime/headers.rs` (new file or add to existing)

```rust
/// Bind Headers constructor and prototype methods
pub fn bind_headers_api(
    scope: &mut v8::ContextScope<v8::HandleScope>,
    context: v8::Local<v8::Context>,
) {
    let global = context.global(scope);
    
    // Create Headers constructor
    let headers_key = v8::String::new(scope, "Headers").unwrap();
    let headers_ctor = v8::Function::new(scope, headers_constructor_callback).unwrap();
    
    // Add prototype methods
    if let Some(headers_obj) = headers_ctor.to_object(scope) {
        let prototype_key = v8::String::new(scope, "prototype").unwrap();
        let prototype = v8::Object::new(scope);
        
        // Bind methods
        bind_headers_method(scope, prototype, "get", headers_get_callback);
        bind_headers_method(scope, prototype, "set", headers_set_callback);
        bind_headers_method(scope, prototype, "has", headers_has_callback);
        bind_headers_method(scope, prototype, "delete", headers_delete_callback);
        bind_headers_method(scope, prototype, "entries", headers_entries_callback);
        bind_headers_method(scope, prototype, "keys", headers_keys_callback);
        bind_headers_method(scope, prototype, "values", headers_values_callback);
        bind_headers_method(scope, prototype, "forEach", headers_foreach_callback);
        
        headers_obj.set(scope, prototype_key.into(), prototype.into());
    }
    
    global.set(scope, headers_key.into(), headers_ctor.into());
}
```

**New (Headers instance):**
```rust
// Create Headers instance
let headers_key = v8::String::new(context_scope, "Headers").unwrap();
let headers_ctor = global.get(context_scope, headers_key.into())
    .and_then(|h| h.to_object(context_scope))
    .and_then(|h| h.cast::<v8::Function>().new_instance(context_scope, &[]));

// Populate headers
if let Some(headers_obj) = headers_ctor {
    handler_ctx.request.headers().for_each(|name, value| {
        // Call headers.set(name, value)
        let set_key = v8::String::new(context_scope, "set").unwrap();
        if let Some(set_fn) = headers_obj.get(context_scope, set_key.into()) {
            if set_fn.is_function() {
                let set = set_fn.cast::<v8::Function>();
                let name_str = v8::String::new(context_scope, name).unwrap();
                let value_str = v8::String::new(context_scope, value).unwrap();
                let _ = set.call(context_scope, headers_obj.into(), &[name_str.into(), value_str.into()]);
            }
        }
    });
    let _ = init_obj.set(context_scope, headers_key.into(), headers_obj.into());
}
```
