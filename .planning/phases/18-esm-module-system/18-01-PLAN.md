1. **Namespace extraction**:
```rust
fn extract_default_export(
    scope: &mut v8::ContextScope<v8::HandleScope>,
    module: v8::Local<v8::Module>,
) -> Result<(Option<v8::Local<v8::Function>>, Option<v8::Local<v8::Object>>)> {
    let namespace = module.get_module_namespace();
    let obj = namespace.to_object(scope)
        .ok_or_else(|| anyhow!("Module namespace is not an object"))?;
    
    // Get 'default' export
    let default_key = v8::String::new(scope, "default").unwrap();
    let default_val = obj.get(scope, default_key.into())
        .ok_or_else(|| anyhow!("No default export found"))?;
    
    // Check if default is an object with fetch method
    if let Some(default_obj) = default_val.to_object(scope) {
        let fetch_key = v8::String::new(scope, "fetch").unwrap();
        let fetch_val = default_obj.get(scope, fetch_key.into());
        
        if let Some(fetch_fn) = fetch_val {
            if fetch_fn.is_function() {
                return Ok((Some(fetch_fn.cast::<v8::Function>()), Some(default_obj)));
            }
        }
    }
    
    // Check if default is directly a function (export default function)
    if default_val.is_function() {
        return Ok((Some(default_val.cast::<v8::Function>()), None));
    }
    
    Err(anyhow!("Default export must be an object with fetch method or a function"))
}
```

```rust
fn resolve_promise_if_needed(
    scope: &mut v8::ContextScope<v8::HandleScope>,
    value: v8::Local<v8::Value>,
) -> Result<v8::Local<v8::Value>> {
    if !value.is_promise() {
        return Ok(value);
    }
    
    let promise = value.cast::<v8::Promise>();
    
    match promise.state() {
        v8::PromiseState::Fulfilled => {
            Ok(promise.result(scope))
        }
        v8::PromiseState::Rejected => {
            let error = promise.result(scope);
            let error_str = error.to_string(scope)
                .map(|s| s.to_rust_string_lossy(scope))
                .unwrap_or_else(|| "Unknown error".to_string());
            Err(anyhow!("Promise rejected: {}", error_str))
        }
        v8::PromiseState::Pending => {
            // Run microtasks to try to settle the promise
            scope.perform_microtask_checkpoint();
            
            // Check again
            match promise.state() {
                v8::PromiseState::Fulfilled => Ok(promise.result(scope)),
                v8::PromiseState::Rejected => {
                    let error = promise.result(scope);
                    let error_str = error.to_string(scope)
                        .map(|s| s.to_rust_string_lossy(scope))
                        .unwrap_or_else(|| "Unknown error".to_string());
                    Err(anyhow!("Promise rejected: {}", error_str))
                }
                v8::PromiseState::Pending => {
                    Err(anyhow!("Promise still pending - async execution incomplete"))
                }
            }
        }
    }
}
```
