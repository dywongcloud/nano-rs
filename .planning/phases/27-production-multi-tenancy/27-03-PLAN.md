   ```rust
   pub struct TenantMetricsCollector {
       tenants: DashMap<String, Arc<RwLock<TenantMetrics>>>,
       global: Arc<RwLock<GlobalMetrics>>,
   }
   
   impl TenantMetricsCollector {
       pub fn record_request(&self, 
           hostname: &str,
           result: RequestResult,
           cpu_time_us: u64,
           memory_bytes: usize,
           duration_ms: u64
       ) {
           let metrics = self.get_or_create(hostname);
           let mut m = metrics.write();
           
           // Update counters
           m.requests_total.inc();
           m.cpu_seconds_total.inc_by(cpu_time_us as f64 / 1_000_000.0);
           
           // Update result counters
           match result {
               RequestResult::Success => m.requests_success.inc(),
               RequestResult::Error(_) => m.requests_error.inc(),
               RequestResult::Timeout => m.requests_timeout.inc(),
           }
           
           // Update histograms
           m.request_duration_seconds.observe(duration_ms as f64 / 1000.0);
           m.cpu_time_per_request_seconds.observe(cpu_time_us as f64 / 1_000_000.0);
           m.memory_per_request_bytes.observe(memory_bytes as f64);
       }
       
       pub fn update_memory(&self, hostname: &str, heap_bytes: usize, external_bytes: usize) {
           let metrics = self.get_or_create(hostname);
           let mut m = metrics.write();
           m.memory_used_bytes.set(heap_bytes as f64);
           m.memory_external_bytes.set(external_bytes as f64);
       }
       
       pub fn record_context_reset(&self, hostname: &str) {
           let metrics = self.get_or_create(hostname);
           metrics.write().context_resets_total.inc();
       }
       
       fn get_or_create(&self, hostname: &str) -> Arc<RwLock<TenantMetrics>> {
           self.tenants
               .entry(hostname.to_string())
               .or_insert_with(|| Arc::new(RwLock::new(TenantMetrics::new(hostname))))
               .clone()
       }
   }
   ```

   ```rust
   fn execute_with_metrics(
       context_manager: &mut ContextManager,
       handler_ctx: &HandlerContext,
       cpu_tracker: &mut CpuTracker,  // From 27-01
       memory_monitor: &MemoryMonitor, // From 27-02
   ) -> Result<NanoResponse> {
       use crate::metrics::TENANT_METRICS;
       
       // Record start
       let request_start = Instant::now();
       let cpu_start = cpu_tracker.snapshot();
       let memory_before = memory_monitor.current_snapshot();
       
       // Execute handler
       let result = execute_with_context_manager(context_manager, handler_ctx);
       
       // Record metrics
       let duration_ms = request_start.elapsed().as_millis() as u64;
       let cpu_us = cpu_tracker.elapsed_since(&cpu_start);
       let memory_after = memory_monitor.check_after(context_manager.isolate_mut().isolate());
       
       let result_type = match &result {
           Ok(_) => RequestResult::Success,
           Err(e) if e.to_string().contains("timeout") => RequestResult::Timeout,
           Err(_) => RequestResult::Error(result.as_ref().err().cloned()),
       };
       
       // Record per-tenant metrics
       TENANT_METRICS.record_request(
           &handler_ctx.hostname,
           result_type,
           cpu_us,
           memory_after.heap_used,
           duration_ms
       );
       
       // Update current memory gauge
       TENANT_METRICS.update_memory(
           &handler_ctx.hostname,
           memory_after.heap_used,
           memory_after.external
       );
       
       // Record pressure events if applicable
       if memory_after.pressure_level > MemoryPressureLevel::Normal {
           TENANT_METRICS.record_pressure_event(&handler_ctx.hostname, memory_after.pressure_level);
       }
       
       result
   }
   ```
