pub fn create_context<'s>(scope: &mut HandleScope<'s>) -> v8::Local<'s, v8::Context> {

pub struct ContextScope<'s, 'c> {

3. HandleScope nesting (critical per D-04):
```rust
pub fn execute_script(
    isolate: &mut v8::Isolate,
    code: &str
) -> anyhow::Result<String> {
    // Scope 1: HandleScope for the operation
    let scope = &mut v8::HandleScope::new(isolate);
    let context = create_context(scope);
    
    // Scope 2: ContextScope for context entry
    let scope = &mut v8::ContextScope::new(scope, context);
    
    // Bind console.log
    bind_console_log(scope, context);
    
    // Scope 3: Compile script (temporary)
    let result = {
        let scope = &mut v8::HandleScope::new(scope);
        let code = v8::String::new(scope, code).unwrap();
        let script = v8::Script::compile(scope, code, None).unwrap();
        script.run(scope)
    };
    
    // Convert result
    match result {
        Some(value) => Ok(value.to_rust_string_lossy(scope)),
        None => Err(anyhow!("Script execution failed")),
    }
}
```
