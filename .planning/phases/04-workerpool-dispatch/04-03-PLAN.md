4. Implement ContextManager::reset_context(&mut self) -> Result<Duration, anyhow::Error>:
   ```rust
   pub fn reset_context(&mut self) -> Result<Duration, anyhow::Error> {
       let start = std::time::Instant::now();
       
       // Step 1: Dispose current context (drop Local<Context>)
       // In V8, Local handles are automatically disposed when they go out of scope
       // We just need to drop our reference
       self.current_context = None;
       
       // Step 2: Force microtask queue drain and GC cleanup (optional, may help)
       // Note: v8::Isolate::PerformMicrotaskCheckpoint might be available
       
       // Step 3: Create new context
       let new_context = self.isolate.create_context();
       self.current_context = Some(new_context);
       self.creation_count += 1;
       
       let elapsed = start.elapsed();
       self.total_reset_time_ms += elapsed.as_secs_f64() * 1000.0;
       
       Ok(elapsed)
   }
   ```

5. Implement ContextManager::context(&self) -> Option<v8::Local<v8::Context>>:

   ```rust
       self.current_context

   ```

6. Implement ContextManager::average_reset_time_ms(&self) -> f64:
   ```rust
   pub fn average_reset_time_ms(&self) -> f64 {
       if self.creation_count > 1 {
           self.total_reset_time_ms / (self.creation_count - 1) as f64
       } else {
           0.0
       }
   }
   ```

1. Modify worker thread spawn code:
   ```rust
   thread::spawn(move || {
       // Initialize V8 isolate (thread-local ownership per POOL-05)
       let mut isolate = NanoIsolate::new().expect("Failed to create isolate");
       
       // Create context manager for this worker
       let mut context_manager = ContextManager::new(&mut isolate);
       context_manager.create_initial_context()
           .expect("Failed to create initial context");
       
       tracing::info!("Worker {} initialized with context", worker_id);
       
       // Event loop
       while let Ok(task) = task_rx.recv() {
           // Reset context before each request (POOL-04)
           match context_manager.reset_context() {
               Ok(elapsed) => {
                   let ms = elapsed.as_secs_f64() * 1000.0;
                   if ms > 10.0 {
                       tracing::warn!("Context reset took {:.2}ms (target <10ms)", ms);
                   } else {
                       tracing::debug!("Context reset took {:.2}ms", ms);
                   }
               }
               Err(e) => {
                   tracing::error!("Context reset failed: {}", e);
                   // Still try to process the task - may fail, but we don't drop it
               }
           }
           
           // Get the current context for execution
           let context = match context_manager.context() {
               Some(ctx) => ctx,
               None => {
                   tracing::error!("No context available for worker {}", worker_id);
                   let _ = task.response_tx.send(Err(anyhow!("Context unavailable")));
                   continue;
               }
           };
           
           // Create HandlerContext
           let handler_ctx = HandlerContext {
               entrypoint: task.entrypoint,
               request: task.request,
           };
           
           // Execute handler WITH explicit context (update execute_handler signature)
           let result = execute_handler_with_context(
               &mut isolate,
               context,
               handler_ctx
           );
           
           // Send response back
           let _ = task.response_tx.send(result);
       }
       
       tracing::info!("Worker {} shutting down (avg context reset: {:.2}ms)", 
                     worker_id, context_manager.average_reset_time_ms());
   })
   ```

   ```rust
   #[test]
   fn test_context_reset_performance_requirement() {
       initialize_platform().expect("Failed to initialize V8 platform");
       
       let mut isolate = NanoIsolate::new().expect("Failed to create isolate");
       let mut manager = ContextManager::new(&mut isolate);
       manager.create_initial_context().expect("Failed to create context");
       
       // Warm up
       for _ in 0..10 {
           manager.reset_context().expect("Failed to reset context");
       }
       
       // Measure 50 resets
       let mut times = vec![];
       for _ in 0..50 {
           let elapsed = manager.reset_context().expect("Failed to reset context");
           times.push(elapsed.as_secs_f64() * 1000.0);
       }
       
       let avg: f64 = times.iter().sum::<f64>() / times.len() as f64;
       let max: f64 = times.iter().cloned().fold(0.0, f64::max);
       let p95: f64 = {
           let mut sorted = times.clone();
           sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
           sorted[(sorted.len() as f64 * 0.95) as usize]
       };
       
       println!("Context reset performance:");
       println!("  Average: {:.2}ms", avg);
       println!("  P95: {:.2}ms", p95);
       println!("  Max: {:.2}ms", max);
       
       // POOL-04 requirement: <10ms (relaxed to <15ms for debug builds)
       assert!(avg < 15.0, "Average context reset time {:.2}ms exceeds 15ms threshold", avg);
       assert!(p95 < 20.0, "P95 context reset time {:.2}ms exceeds 20ms threshold", p95);
   }
   ```
