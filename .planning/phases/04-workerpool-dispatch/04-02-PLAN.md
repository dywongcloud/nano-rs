   ```rust
   pub async fn handle_request(
       State(state): State<AppState>,
       req: Request<Body>,
   ) -> Result<Response<Body>, StatusCode> {
       // Extract hostname from request
       let hostname = extract_hostname(&req)?;
       
       // Look up route target
       let target = state.router.route(&hostname);
       
       // Convert axum request to NanoRequest
       let nano_req = convert_to_nano_request(req).await?;
       
       // Create oneshot channel for response
       let (tx, rx) = oneshot::channel();
       
       // Create handler task
       let task = HandlerTask {
           entrypoint: target.entrypoint.clone(),
           request: nano_req,
           response_tx: tx,
       };
       
       // Dispatch to WorkQueue (async Mutex lock)
       let mut queue = state.work_queue.lock().await;
       match queue.dispatch(&hostname, task) {
           Ok(()) => {
               // Wait for response
               match rx.await {
                   Ok(Ok(response)) => convert_to_axum_response(response),
                   Ok(Err(e)) => {
                       tracing::error!("Handler error: {}", e);
                       Ok(Response::builder()
                           .status(500)
                           .body(Body::from("Internal Server Error"))
                           .unwrap())
                   }
                   Err(_) => {
                       tracing::error!("Response channel closed");
                       Ok(Response::builder()
                           .status(500)
                           .body(Body::from("Internal Server Error"))
                           .unwrap())
                   }
               }
           }
           Err(QueueError::ChannelFull) => {
               tracing::warn!("WorkQueue full for hostname: {}", hostname);
               Ok(Response::builder()
                   .status(503)
                   .header("Retry-After", "1")
                   .body(Body::from("Service Unavailable - Queue Full"))
                   .unwrap())
           }
           Err(e) => {
               tracing::error!("Dispatch error: {}", e);
               Ok(Response::builder()
                   .status(500)
                   .body(Body::from("Internal Server Error"))
                   .unwrap())
           }
       }
   }
   ```
