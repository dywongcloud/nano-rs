    //! Integration tests for WASM async execution
    //!
    //! These tests verify that WebAssembly.compile() and WebAssembly.instantiate()
    //! actually resolve their Promises through the async event loop.

    use nano_rs::worker::pool::{SliverWorkerPool, WorkerPool};
    use nano_rs::http::{NanoHeaders, NanoRequest, NanoUrl};
    use nano_rs::vfs::{MemoryBackend, VfsNamespace};
    use std::time::Duration;

    /// Add WASM module bytes (3 + 5 = 8)
    fn add_wasm_bytes() -> Vec<u8> {
        vec![
            0x00, 0x61, 0x73, 0x6d, 0x01, 0x00, 0x00, 0x00,
            0x01, 0x07, 0x01, 0x60, 0x02, 0x7f, 0x7f, 0x01, 0x7f,
            0x03, 0x02, 0x01, 0x00,
            0x07, 0x08, 0x01, 0x03, 0x61, 0x64, 0x64, 0x00, 0x00,
            0x0a, 0x09, 0x01, 0x07, 0x00,
            0x20, 0x00, 0x20, 0x01, 0x6a, 0x0b,
        ]
    }

    /// Test helper: Create pool with WASM handler
    async fn setup_wasm_test() -> (SliverWorkerPool, tempfile::TempDir) {
        let temp_dir = tempfile::tempdir().unwrap();
        let base_path = temp_dir.path().to_path_buf();
        
        // Write WASM file
        let wasm_path = base_path.join("add.wasm");
        tokio::fs::write(&wasm_path, add_wasm_bytes()).await.unwrap();
        
        // Create handler.js that uses WASM
        let handler_js = r#"
            export default {
                async fetch(request) {
                    const wasmBytes = await Nano.fs.readFile('./add.wasm');
                    const module = await WebAssembly.compile(wasmBytes);
                    const instance = await WebAssembly.instantiate(module, {});
                    const result = instance.exports.add(5, 3);
                    return new Response(String(result), {
                        headers: { 'Content-Type': 'text/plain' }
                    });
                }
            };
        "#;
        let handler_path = base_path.join("handler.js");
        tokio::fs::write(&handler_path, handler_js).await.unwrap();
        
        // Create worker pool
        let vfs = IsolateVfs::new(Box::new(MemoryBackend::new()), VfsNamespace::new("test"));
        let pool = SliverWorkerPool::new(
            1, // Single worker for deterministic tests
            128,
            handler_path.to_str().unwrap(),
            vfs,
        ).await.expect("Failed to create pool");
        
        (pool, temp_dir)
    }

    #[tokio::test]
    async fn test_wasm_compile_resolves() {
        // Before fix: Returns "Promise still pending"
        // After fix: Returns compiled module that can be instantiated
        let (pool, _temp) = setup_wasm_test().await;
        
        let request = NanoRequest::new(
            "GET".to_string(),
            NanoUrl::parse("http://test.wasm/compile").unwrap(),
            NanoHeaders::new(),
            None,
        );
        
        let response = pool.execute(request, Duration::from_secs(10)).await
            .expect("Pool execution should succeed");
        
        // Should NOT be 500 with "Promise still pending"
        assert_ne!(response.status, 500, "Should not return error status");
        
        let body_str = String::from_utf8_lossy(&response.body.unwrap_or_default());
        assert_ne!(body_str, "Promise still pending", 
            "Promise should have resolved through async loop");
        assert_ne!(body_str, "Promise still pending - async execution not fully supported",
            "Should not see the old error message");
    }
