fn test_concurrent_sliver_creation() {
    let temp_dir = Arc::new(TempDir::new().unwrap());
    let mut handles = vec![];
    
    // Spawn 10 threads creating slivers simultaneously
    for i in 0..10 {
        let temp = Arc::clone(&temp_dir);
        let handle = thread::spawn(move || {
            let sliver_path = temp.path().join(format!("concurrent-{}.sliver", i));
            
            // Each thread creates its own sliver
            create_test_sliver(&sliver_path, &format!("app{}.example.com", i));
            
            // Verify it was created
            assert!(sliver_path.exists());

        handles.push(handle);
    }
    
    // Wait for all threads
    for handle in handles {
        handle.join().unwrap();
    }
    
    // Verify all slivers exist
    for i in 0..10 {
        let path = temp_dir.path().join(format!("concurrent-{}.sliver", i));
        assert!(path.exists(), "Sliver {} should exist", i);

fn test_concurrent_list_during_creation() {
    let temp_dir = Arc::new(TempDir::new().unwrap());
    let creating = Arc::new(std::sync::atomic::AtomicBool::new(true));
    
    // Thread 1: Continuously create and delete slivers
    let temp1 = Arc::clone(&temp_dir);
    let creating1 = Arc::clone(&creating);
    let creator = thread::spawn(move || {
        for i in 0..50 {
            let path = temp1.path().join(format!("flappy-{}.sliver", i));
            create_test_sliver(&path, &format!("flappy{}.example.com", i));
            thread::sleep(Duration::from_millis(5));
            std::fs::remove_file(&path).ok();
        }
        creating1.store(false, std::sync::atomic::Ordering::SeqCst);
    });
    
    // Thread 2: Continuously list slivers
    let temp2 = Arc::clone(&temp_dir);
    let creating2 = Arc::clone(&creating);
    let lister = thread::spawn(move || {
        while creating2.load(std::sync::atomic::Ordering::SeqCst) {
            // List slivers in directory
            let entries: Vec<_> = std::fs::read_dir(temp2.path())
                .unwrap()
                .filter_map(|e| e.ok())
                .filter(|e| {
                    e.path().extension()
                        .map(|ext| ext == "sliver")
                        .unwrap_or(false)
                })
                .collect();
            
            // Should not panic even with concurrent modifications
            thread::sleep(Duration::from_millis(5));
        }
    });
    
    creator.join().unwrap();
    lister.join().unwrap();
}

pub fn assert_sliver_valid(path: &Path) {
    let data = std::fs::read(path).expect("Failed to read sliver");
    let cursor = std::io::Cursor::new(data);
    let mut archive = tar::Archive::new(cursor);
    
    let mut found_metadata = false;
    let mut found_heap = false;
    
    for entry in archive.entries().expect("Failed to read archive") {
        let entry = entry.expect("Invalid entry");
        let path = entry.path().expect("Invalid path").to_string_lossy().to_string();
        
        if path == "metadata.json" {
            found_metadata = true;
        } else if path == "heap.bin" {
            found_heap = true;
        }
    }
    
    assert!(found_metadata, "Missing metadata.json");
    assert!(found_heap, "Missing heap.bin");
}
