```rust
pub fn load_sliver_with_search(name_or_path: &str) -> Result<SliverData, CliError> {
    // Try as direct path first
    let path = Path::new(name_or_path);
    if path.exists() {
        return load_sliver(path);
    }
    
    // Search in common locations
    let search_paths = vec![
        format!("./{}.sliver", name_or_path),
        format!("{}/{}.sliver", std::env::var("NANO_SLIVER_DIR").unwrap_or("./slivers".into()), name_or_path),
        format!("{}/.nano/slivers/{}.sliver", dirs::home_dir().unwrap_or_default().display(), name_or_path),
    ];
    
    for path_str in &search_paths {
        let path = Path::new(path_str);
        if path.exists() {
            return load_sliver(path);
        }
    }
    
    // Check if it's a typo of an existing sliver
    let existing = list_slivers_in_paths(&search_paths);
    if let Some(suggestion) = find_similar(name_or_path, &existing) {
        return Err(CliError::SliverNotFoundWithSuggestion {
            name: name_or_path.to_string(),
            suggestion,
            searched: search_paths.into_iter().map(PathBuf::from).collect(),
        });
    }
    
    Err(CliError::SliverNotFound {
        name: name_or_path.to_string(),
        searched: search_paths.into_iter().map(PathBuf::from).collect(),
    })
}
```

**Acceptance Criteria:**

```rust
pub enum CorruptionType {
    InvalidTar { reason: String },
    MissingMetadata,
    MissingHeap,
    InvalidMetadata { error: String },
    TruncatedFile { expected: u64, found: u64, entry: String },
    ChecksumMismatch { entry: String },
}

pub fn validate_sliver_integrity(path: &Path) -> Result<(), CorruptionError> {
    let file = File::open(path)?;
    let mut archive = Archive::new(file);
    
    let mut found_metadata = false;
    let mut found_heap = false;
    
    for entry in archive.entries()? {
        let entry = entry?;
        let path = entry.path()?;
        let size = entry.size();
        
        match path.to_str() {
            Some("metadata.json") => {
                found_metadata = true;
                validate_metadata_entry(&mut entry)?;
            }
            Some("heap.bin") => {
                found_heap = true;
                if size == 0 {
                    return Err(CorruptionType::TruncatedFile {
                        expected: 1, // Any non-zero
                        found: 0,
                        entry: "heap.bin".into(),
                    });
                }
            }
            Some(path) if path.starts_with("vfs/") => {
                validate_vfs_entry(&mut entry)?;
            }
            _ => {}
        }
    }
    
    if !found_metadata {
        return Err(CorruptionType::MissingMetadata);
    }
    if !found_heap {
        return Err(CorruptionType::MissingHeap);
    }
    
    Ok(())
}

fn validate_metadata_entry<R: Read>(entry: &mut tar::Entry<R>) -> Result<(), CorruptionType> {
    let mut content = String::new();
    entry.read_to_string(&mut content)
        .map_err(|e| CorruptionType::InvalidMetadata { error: e.to_string() })?;
    
    // Parse as JSON to validate structure
    let _: SliverMetadata = serde_json::from_str(&content)
        .map_err(|e| CorruptionType::InvalidMetadata { error: e.to_string() })?;
    
    Ok(())
}
```

**Acceptance Criteria:**
- Detects missing required entries
- Validates metadata JSON structure
- Detects truncated files
- Reports specific corruption type
- Can be called before full unpacking

```rust
pub const SUPPORTED_FORMAT_VERSIONS: &[&str] = &["1.0"];
pub const CURRENT_FORMAT_VERSION: &str = "1.0";

#[derive(Debug)]
pub enum VersionError {
    TooNew {
        sliver_version: String,
        supported: Vec<String>,
    },
    TooOld {
        sliver_version: String,
        supported: Vec<String>,
        migration_available: bool,
    },
}

pub fn check_version_compatibility(metadata: &SliverMetadata) -> Result<(), VersionError> {
    let sliver_version = &metadata.format_version;
    
    if SUPPORTED_FORMAT_VERSIONS.contains(&sliver_version.as_str()) {
        return Ok(());
    }
    
    // Parse versions for comparison
    let sliver_major = parse_major_version(sliver_version);
    let current_major = parse_major_version(CURRENT_FORMAT_VERSION);
    
    if sliver_major > current_major {
        Err(VersionError::TooNew {
            sliver_version: sliver_version.clone(),
            supported: SUPPORTED_FORMAT_VERSIONS.iter().map(|s| s.to_string()).collect(),
        })
    } else {
        // Check if migration is available
        let migration_available = check_migration_available(sliver_version);
        
        Err(VersionError::TooOld {
            sliver_version: sliver_version.clone(),
            supported: SUPPORTED_FORMAT_VERSIONS.iter().map(|s| s.to_string()).collect(),
            migration_available,
        })
    }
}

impl fmt::Display for VersionError {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        match self {
            VersionError::TooNew { sliver_version, supported } => {
                writeln!(f, "✗ Sliver format version {} is too new", sliver_version)?;
                writeln!(f)?;
                writeln!(f, "This sliver was created with a newer version of nano-rs.")?;
                writeln!(f, "Supported versions: {}", supported.join(", "))?;
                writeln!(f)?;
                write!(f, "Please upgrade nano-rs to open this sliver.")
            }
            VersionError::TooOld { sliver_version, supported, migration_available } => {
                writeln!(f, "✗ Sliver format version {} is outdated", sliver_version)?;
                writeln!(f)?;
                writeln!(f, "Supported versions: {}", supported.join(", "))?;
                if *migration_available {
                    writeln!(f)?;
                    write!(f, "Run 'nano-rs sliver migrate <path>' to upgrade this sliver.")
                } else {
                    write!(f, "Please recreate this sliver with the current version.")
                }
            }
        }
    }
}
```

**Acceptance Criteria:**
- Detects version too new (needs upgrade)
- Detects version too old (needs migration or recreate)
- Human-readable error messages
- Suggests migration command if available
- Version constants are maintainable

```rust
pub fn check_snapshot_compatibility(metadata: &SliverMetadata) -> Result<(), SnapshotCompatibilityError> {
    let runtime_v8_version = get_runtime_v8_version();
    let snapshot_v8_version = metadata.v8_version.as_deref()
        .ok_or(SnapshotCompatibilityError::MissingVersion)?;
    
    // V8 snapshots are generally compatible within major versions
    let runtime_major = extract_major(&runtime_v8_version);
    let snapshot_major = extract_major(snapshot_v8_version);
    
    if runtime_major != snapshot_major {
        return Err(SnapshotCompatibilityError::MajorVersionMismatch {
            runtime: runtime_v8_version,
            snapshot: snapshot_v8_version.to_string(),
        });
    }
    
    // Warn on minor version differences (soft error)
    let runtime_minor = extract_minor(&runtime_v8_version);
    let snapshot_minor = extract_minor(snapshot_v8_version);
    
    if runtime_minor != snapshot_minor {
        log::warn!(
            "V8 minor version mismatch: runtime={}, snapshot={}. \
             Snapshot may work but could have subtle issues.",
            runtime_v8_version, snapshot_v8_version
        );
    }
    
    Ok(())
}

pub fn get_runtime_v8_version() -> String {
    // Get V8 version from rusty_v8
    format!("{}", v8::V8::get_version())
}
```

**Acceptance Criteria:**
- Checks V8 major version compatibility
- Warns on minor version differences
- Returns error on major mismatch
- Graceful handling if version missing from metadata

```rust
pub async fn restore_with_fallback(
    sliver: &UnpackedSliver,
    entrypoint: Option<&Path>,
) -> Result<RestoredIsolate, RestoreError> {
    // Try snapshot restoration first
    match restore_from_snapshot(sliver).await {
        Ok(isolate) => {
            log::info!("✓ Restored isolate from snapshot");
            return Ok(isolate);
        }
        Err(e) => {
            log::warn!("Snapshot restoration failed: {}. Falling back to fresh isolate.", e);
        }
    }
    
    // Fallback: create fresh isolate
    if let Some(entry) = entrypoint {
        match create_fresh_isolate(entry, &sliver.metadata).await {
            Ok(isolate) => {
                log::info!("✓ Created fresh isolate (snapshot unavailable)");
                Ok(isolate)
            }
            Err(e) => {
                Err(RestoreError::FallbackFailed {
                    snapshot_error: Box::new(e),
                    fresh_error: Box::new(e),
                })
            }
        }
    } else {
        Err(RestoreError::NoFallbackAvailable {
            reason: "No entrypoint specified for fresh isolate creation".into(),
        })
    }
}

// For CLI, show user-friendly message about fallback
pub fn format_fallback_message(result: &RestoredIsolate) -> String {
    match result.source {
        RestoreSource::Snapshot => "Restored from snapshot (fast cold start)".into(),
        RestoreSource::Fallback => "Created fresh isolate (snapshot unavailable)".into(),
    }
}
```

**Acceptance Criteria:**
- Automatically falls back to fresh isolate
- Logs warning about snapshot failure
- Still restores VFS state even without snapshot
- Clear messaging to user about fallback
- Respects --strict flag to disable fallback
