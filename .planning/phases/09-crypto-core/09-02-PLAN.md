From ring crate documentation (per research):
```rust
use ring::aead::{Aes128Gcm, Aes256Gcm, Nonce, UnboundKey, AES_128_GCM, AES_256_GCM};

// Key generation: random bytes
let key_bytes: [u8; 32] = generate_random_bytes();

// Encryption
let key = UnboundKey::new(&AES_256_GCM, &key_bytes)?;
let nonce = Nonce::try_assume_unique_for_key(&iv_bytes)?;
let mut in_out = plaintext_bytes;
let tag = seal_in_place_separate_tag(
    key,
    nonce,
    Aad::empty(),
    &mut in_out,
    AES_256_GCM.tag_len(),
)?;

// Decryption
open_in_place(key, nonce, Aad::empty(), 0.., &mut in_out)?;
```

WebCrypto AES-GCM algorithm parameters:
```javascript
{
  name: "AES-GCM",
  iv: Uint8Array,           // 12 bytes recommended
  additionalData: Uint8Array, // optional
  tagLength: 128            // 32, 64, 96, 104, 112, 120, or 128
}
```

JWK format for AES keys:
```json
{
  "kty": "oct",
  "k": "base64url-encoded-key",
  "alg": "A256GCM",
  "ext": true,
  "key_ops": ["encrypt", "decrypt"]
}
```
