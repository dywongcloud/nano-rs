From ring crate documentation (per research):
```rust
use ring::hmac::{Key, HMAC_SHA256, sign, verify};

// Key generation
let key_bytes: [u8; 32] = generate_random_bytes();
let key = Key::new(HMAC_SHA256, &key_bytes);

// Sign
let signature = sign(&key, message);
let signature_bytes = signature.as_ref();

// Verify
let result = verify(&key, message, signature_bytes);
// Returns Ok(()) on success, Err(_) on failure
```

```javascript
{
  name: "HMAC",
  hash: "SHA-256",       // Required: SHA-256, SHA-384, SHA-512
```

```json
{
  "kty": "oct",
  "k": "base64url-encoded-key",
  "alg": "HS256",        // HS256, HS384, HS512
  "ext": true,
  "key_ops": ["sign", "verify"]
}
```
