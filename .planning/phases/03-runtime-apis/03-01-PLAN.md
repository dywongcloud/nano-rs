    ```rust
    fn create_js_request(
        scope: &mut v8::HandleScope,
        request: &NanoRequest,
    ) -> anyhow::Result<v8::Local<v8::Object>> {
        // Serialize NanoRequest to JSON using existing v8_bridge pattern
        let json_str = crate::http::v8_bridge::serialize_request_to_json(request);
        
        // Parse JSON in V8 context to create JavaScript object
        let json_obj = v8::String::new(scope, &json_str)
            .ok_or_else(|| anyhow!("Failed to create JSON string"))?;
        
        // Use JSON.parse to convert to JS object
        let parse_key = v8::String::new(scope, "JSON").unwrap();
        let global = scope.get_current_context().global(scope);
        let json_obj = global.get(scope, parse_key.into())
            .ok_or_else(|| anyhow!("JSON not found in global"))?;
        let json_obj = json_obj.to_object(scope)
            .ok_or_else(|| anyhow!("JSON is not an object"))?;
        
        let parse_key = v8::String::new(scope, "parse").unwrap();
        let parse_fn = json_obj.get(scope, parse_key.into())
            .ok_or_else(|| anyhow!("JSON.parse not found"))?
            .cast::<v8::Function>();
        
        // Call JSON.parse(json_str) to get the Request object
        let result = parse_fn.call(scope, json_obj.into(), &[json_obj.into()]);
        
        result.ok_or_else(|| anyhow!("Failed to parse request JSON"))?
            .to_object(scope)
            .ok_or_else(|| anyhow!("Parsed result is not an object"))
    }
    ```

    ```rust
        scope: &mut v8::HandleScope,
        js_response: v8::Local<v8::Value>,

        if let Some(headers_obj) = headers_val.to_object(scope) {

        // Extract body property (may be string, Uint8Array, or null)
        let body_key = v8::String::new(scope, "body").unwrap();
    ```

    ```rust
    pub async fn handle(&self, request: NanoRequest) -> NanoResponse {
        match &self.handler_type {
            HandlerType::StaticResponse(response) => {
                NanoResponse::ok()
                    .with_header("Content-Type", "text/plain")
                    .with_body(response.clone())
            }
            HandlerType::WinterTCHandler(entrypoint) => {
                // Execute JavaScript handler (Phase 3)
                use crate::runtime::{HandlerContext, execute_handler};
                use crate::v8::NanoIsolate;
                
                let context = HandlerContext {
                    entrypoint: entrypoint.clone(),
                    request,
                };
                
                // Create a fresh isolate for this request (Phase 4 will optimize)
                match NanoIsolate::new() {
                    Ok(mut isolate) => {
                        match execute_handler(&mut isolate, context).await {
                            Ok(response) => response,
                            Err(e) => {
                                tracing::error!("Handler execution failed: {}", e);
                                NanoResponse::with_status(500)
                                    .with_header("Content-Type", "text/plain")
                                    .with_body("Internal Server Error")
                            }
                        }
                    }
                    Err(e) => {
                        tracing::error!("Failed to create isolate: {}", e);
                        NanoResponse::with_status(500)
                            .with_header("Content-Type", "text/plain")
                            .with_body("Failed to initialize runtime")
                    }
                }
            }
        }
    }
    ```
