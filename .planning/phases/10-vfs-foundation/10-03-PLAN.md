Layer 1: PathValidator (security.rs)
- Strictest validation, rejects anything suspicious
- Called before normalization

Layer 2: VfsPath::new() (types.rs)
- Normalization and basic validation
- Rejects ".." segments

Layer 3: IsolateVfs (isolate.rs)
- Namespace prefixing
- Additional path check before storage access

Layer 4: Backend (memory.rs)
- Storage-level isolation via keys
- Last line of defense

## Resource Limits

Per-namespace limits (configurable):
- Max file size: 10MB default (protects against DoS)
- Max total storage: 100MB default per namespace
- Max files: 1000 default per namespace
- Max path length: 4096 bytes

   ```rust
   impl PathValidator {
       pub fn new(max_path_length: usize) -> Self {
           Self { max_path_length, ..Default::default() }
       }
       
       /// Strict validation - rejects anything suspicious
       pub fn validate(&self, path: &str) -> Result<(), VfsError> {
           // Check length
           if path.len() > self.max_path_length {
               return Err(VfsError::InvalidPath(
                   format!("Path exceeds max length: {}", self.max_path_length)
               ));
           }
           
           // Check for null bytes
           if path.contains('\0') {
               return Err(VfsError::InvalidPath(
                   "Path contains null bytes".to_string()
               ));
           }
           
           // Check for traversal attempts (before normalization)
           if path.contains("..") {
               return Err(VfsError::InvalidPath(
                   "Path contains '..' which is not allowed".to_string()
               ));
           }
           
           // Check for absolute paths (rejected for VFS paths)
           if path.starts_with('/') {
               // This is actually OK after normalization, but raw input shouldn't start with /
               // Log warning in debug mode
               tracing::debug!("Path starts with /: {}", path);
           }
           
           Ok(())
       }
   }
   ```

   ```rust
   async fn write(&self, path: &VfsPath, content: &[u8]) -> VfsResult<()> {
       let content_len = content.len();
       
       // Check file size limit
       if content_len > self.limits.max_file_size {
           return Err(VfsError::QuotaExceeded {
               resource: "file_size".to_string(),
               limit: self.limits.max_file_size,
               current: content_len,
           });
       }
       
       // Check if this is a new file or update
       let is_new = !self.storage.contains_key(path.as_str());
       
       if is_new {
           // Check file count limit
           let current_count = self.file_count.load(Ordering::SeqCst);
           if current_count >= self.limits.max_files {
               return Err(VfsError::QuotaExceeded {
                   resource: "file_count".to_string(),
                   limit: self.limits.max_files,
                   current: current_count,
               });
           }
       }
       
       // Calculate new total size
       let old_size = self.storage
           .get(path.as_str())
           .map(|entry| entry.content.len())
           .unwrap_or(0);
       
       let size_delta = content_len as i64 - old_size as i64;
       let current_total = self.total_bytes.load(Ordering::SeqCst) as i64;
       let new_total = (current_total + size_delta) as usize;
       
       // Check total storage limit
       if new_total > self.limits.max_total_storage {
           return Err(VfsError::QuotaExceeded {
               resource: "total_storage".to_string(),
               limit: self.limits.max_total_storage,
               current: current_total as usize,
           });
       }
       
       // Store the file
       let file = VfsFile {
           content: content.to_vec(),
           created_at: if is_new { 
               std::time::SystemTime::now() 
           } else { 
               self.storage.get(path.as_str()).unwrap().created_at 
           },
           modified_at: std::time::SystemTime::now(),
           size: content_len,
       };
       
       self.storage.insert(path.as_str().to_string(), file);
       
       // Update counters
       if is_new {
           self.file_count.fetch_add(1, Ordering::SeqCst);
       }
       self.total_bytes.store(new_total, Ordering::SeqCst);
       
       Ok(())
   }
   ```
