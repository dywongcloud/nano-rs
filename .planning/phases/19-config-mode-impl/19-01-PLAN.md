```rust
async fn run_server_with_config(config_path: PathBuf) -> Result<()> {
    // Load and validate config (existing)
    let config = nano::config::load_config(&config_path)?;
    tracing::info!("Loaded configuration with {} app(s)", config.apps.len());
    
    // Initialize V8 platform
    nano::v8::platform::initialize_platform()
        .context("Failed to initialize V8 platform")?;
    
    // Create app registry from config
    let registry = Arc::new(RwLock::new(nano::app::registry::AppRegistry::from_config(config.clone())));
    
    // Create worker pools per app
    let mut worker_pools: HashMap<String, Arc<dyn AppWorkerPool>> = HashMap::new();
    
    for app in &config.apps {
        if let Some(ref sliver_path) = app.sliver {
            // Sliver-based app
            let sliver_data = std::fs::read(sliver_path)?;
            let unpacked = nano::sliver::unpack_sliver(&sliver_data)?;
            let pool = Arc::new(nano::worker::SliverWorkerPool::new(
                app.hostname.clone(),
                app.limits.workers,
                app.limits.memory_mb as usize * 1024 * 1024, // Convert MB to bytes
                unpacked,
            ));
            worker_pools.insert(app.hostname.clone(), pool);
        } else {
            // Entrypoint-based app (simpler pool)
            let pool = Arc::new(nano::worker::EntrypointWorkerPool::new(
                app.hostname.clone(),
                app.entrypoint.clone(),
                app.limits.workers,
                app.limits.memory_mb as usize * 1024 * 1024,
            ));
            worker_pools.insert(app.hostname.clone(), pool);
        }
    }
    
    // Set up graceful shutdown
    let drain = nano::app::drain::RequestDrain::new();
    let (shutdown, mut shutdown_rx) = nano::signal::setup_shutdown(
        nano::signal::ShutdownConfig::default(),
        drain,
    );
    
    // Convert server config section to ServerConfig
    let server_bind_config = nano::http::ServerConfig::from(config.server.clone());
    tracing::info!("Starting HTTP server on {}", server_bind_config.socket_addr()?);
    
    // Start server with config
    let shutdown_state = shutdown.state().clone();
    let server_handle = tokio::spawn(async move {
        nano::http::start_server_with_pools(
            server_bind_config,
            worker_pools,
            registry,
            shutdown_state,
        ).await
    });
    
    // Wait for shutdown signal
    let _ = shutdown_rx.recv().await;
    tracing::info!("Shutdown signal received...");
    
    // Graceful shutdown
    shutdown.shutdown().await;
    
    // Shutdown worker pools
    for (hostname, pool) in worker_pools.iter() {
        tracing::info!("Shutting down worker pool for {}", hostname);
        if let Ok(pool) = Arc::try_unwrap(Arc::clone(pool)) {
            let _ = pool.shutdown();
        }
    }
    
    // Wait for server
    match server_handle.await {
        Ok(Ok(())) => tracing::info!("Server stopped successfully"),
        Ok(Err(e)) => tracing::error!("Server error: {}", e),
        Err(e) => tracing::error!("Server task panicked: {}", e),
    }
    
    tracing::info!("NANO config mode shutdown complete");
    Ok(())
}
```

**Note:** If `EntrypointWorkerPool` doesn't exist, we'll need to either:
- Create it (adds scope)
- Or only support sliver-based apps in config mode for now

Given time constraints, support sliver-based apps first, then add entrypoint support.

       State(state): State<Arc<MultiAppState>>,
       req: Request,
   ) -> Response {
       // Extract Host header
       let hostname = req.headers()
           .get("host")
           .and_then(|h| h.to_str().ok())
           .unwrap_or("default");
       
       // Lookup pool for hostname
       if let Some(pool) = state.worker_pools.get(hostname) {
           // Convert axum request to NanoRequest

                   StatusCode::INTERNAL_SERVER_ERROR.into_response()

# Config Mode

NANO supports running multiple applications from a JSON configuration file.

- `entrypoint` (optional): Path to JS entrypoint file (not yet supported in config mode)

- `limits`: Resource limits for this app

  - `timeout_secs`: Request timeout in seconds (1-300)
  - `workers`: Number of worker threads (1-32)

### Server Configuration

- `port`: Port to bind (default: 8080)
- `host`: Bind address (default: "0.0.0.0")

## Virtual Host Routing

- `api.example.com` → routes to API app
- `blog.example.com` → routes to Blog app

| Boundary | Description |
|----------|-------------|
| config file → runtime | Config file is trusted input (administrator-controlled) but must be validated |
| request → worker pool | Requests from any client must be isolated to their app's pool |
| app A → app B | Apps must not share worker pools or memory |

| T-19-02 | Elevation of Privilege | Worker pool isolation | mitigate | Each app gets dedicated worker pool, no cross-app pool sharing |
